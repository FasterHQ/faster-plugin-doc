## Key Files

<CodeGroup>
```javascript index.js
const {BasePlugin} = require('@fasterhq/plugin-sdk');

class BlogExtension extends BasePlugin {
    constructor() {
        super({name: 'blog-extension', version: '1.0.0', pluginPath: __dirname});
    }

    async onInit() {
        this.debug('Blog Extension initialized');
    }
}

module.exports = new BlogExtension();
```

```json manifest.json
{
  "name": "Blog Extension",
  "package_name": "blog-extension",
  "version": "1.0.0",
  "routing": "blog-ext",
  "permissions": ["database"]
}
```
