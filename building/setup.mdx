## Prerequisites

<AccordionGroup>
  <Accordion title="Node.js 18+" icon="node-js">
    ```bash
    node --version  # Should be 18.x or higher
    ```
  </Accordion>

  <Accordion title="Faster CMS" icon="ghost">
    ```bash
    npm install -g ghost-cli
    ghost install local
    ```
  </Accordion>
</AccordionGroup>

## Create Plugin

```bash
cd ghost/core/content/plugins
mkdir my-plugin
cd my-plugin
```

## Essential Files

<Steps>
  <Step title="manifest.json">
    ```json
    {
      "name": "My Plugin",
      "package_name": "my-plugin",
      "version": "1.0.0",
      "routing": "myplugin",
      "permissions": ["admin", "database"]
    }
    ```
  </Step>

  <Step title="index.js">
    ```javascript
    const {BasePlugin} = require('@fasterhq/plugin-sdk');

    class MyPlugin extends BasePlugin {
        constructor() {
            super({name: 'my-plugin', version: '1.0.0', pluginPath: __dirname});
        }

        async onInit() {
            this.debug('Plugin initialized');
        }
    }

    module.exports = new MyPlugin();
    ```
  </Step>

  <Step title="routes.js">
    ```javascript
    const {PluginRouter} = require('@fasterhq/plugin-sdk');
    const router = new PluginRouter('myplugin');

    router.public('get', '/', (req, res) => {
        res.send('<h1>My Plugin</h1>');
    });

    module.exports = {router: () => router.router()};
    ```
  </Step>
</Steps>
