router.admin('put', '/settings', controller.updateSettings);

## Middleware

```javascript routes.js
const bodyParser = require('body-parser');
const router = new PluginRouter('shop');
const expressRouter = router.getExpressRouter();

// Global middleware
expressRouter.use(bodyParser.json());
expressRouter.use(customMiddleware);

// Route-specific
router.public('post', '/api/action', validateMiddleware, controller.action);
```

## Parameters

```javascript
// Path parameters
router.public('get', '/products/:slug', (req, res) => {
    const slug = req.params.slug;
});

// Query parameters
router.public('get', '/search', (req, res) => {
    const {q, page} = req.query;
    // URL: /shop/search?q=laptop&page=2
});

// Request body
router.public('post', '/api/cart/add', (req, res) => {
    const {productId, quantity} = req.body;
});
```
